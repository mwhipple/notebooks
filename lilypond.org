  -d, --define-default=help        show help for Scheme options
  -e, --evaluate=EXPR              evaluate scheme code
  -h, --help                       show this help and exit
  -H, --header=FIELD               dump \header field FIELD to file named
                                     BASENAME.FIELD
  -I, --include=DIR                append DIR to search path
  -i, --init=FILE                  use FILE as init file
